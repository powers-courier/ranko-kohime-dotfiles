**** DONE Proxy Hosts Generator
***** Explanation
****** proxyCount:
Set to 3—generates jelly-proxy-01, jelly-proxy-02, jelly-proxy-03.
Change to 5 for jelly-proxy-05, etc.
****** builtins.genList:
builtins.genList (i: i) 3 → [0 1 2]—a list of indices.
i + 1 shifts it to 1, 2, 3 for numbering.
****** map:
Loops over [0 1 2], creating a list of { name = ...; value = ...; } pairs.
num: Pads with 0 for 01-09 (e.g., if i < 9 then "0${toString (i + 1)}"), then plain numbers (10, 11, ...).
****** builtins.listToAttrs:
Turns the list into an attrset:
****** networking.hostName = name:
Inside the inline module, name (e.g., "jelly-proxy-01") sets the hostName dynamically—no repetition.
****** // Operator:
Merges proxyConfigs with other configs (e.g., jellyfin-db)—keeps non-proxy hosts separate.
***** Code

{
  outputs = { self, nixpkgs, ... }@inputs:
    let
      system = "x86_64-linux"; # Adjust as needed

      # Common hardware configuration shared by all hosts
      commonHardwareConfig = { config, lib, pkgs, modulesPath, ... }: {
        imports = [
          (modulesPath + "/installer/scan/not-detected.nix")
        ];

        boot.initrd.availableKernelModules = [ "xhci_pci" "ahci" "nvme" "usb_storage" "sd_mod" ];
        boot.initrd.kernelModules = [ ];
        boot.kernelModules = [ "kvm-intel" ];
        boot.extraModulePackages = [ ];

        networking.useDHCP = lib.mkDefault true;
        nixpkgs.hostPlatform = lib.mkDefault "x86_64-linux";
        hardware.cpu.intel.updateMicrocode = lib.mkDefault config.hardware.enableRedistributableFirmware;
      };
      # Fallback number of hosts if auto-detection fails
      fallbackHostCount = 10;

      # Helper function to format number with leading zero
      formatHostNumber = num: 
        if num < 10 then "0${toString num}" else toString num;

      # Helper function to generate hostname
      generateHostname = num: "jelly-proxy-${formatHostNumber num}";

      # Auto-detect available hardware configs
      detectAvailableHosts = 
        let
          # Get all available host numbers from hardwareConfigs
          availableHosts = builtins.filter (name: 
            builtins.match "jelly-proxy-[0-9][0-9]" name != null
          ) (builtins.attrNames hardwareConfigs);
          
          # Extract numbers and find the maximum
          hostNumbers = builtins.map (name:
            let
              matches = builtins.match "jelly-proxy-([0-9][0-9])" name;
            in
              if matches != null 
              then builtins.fromJSON (builtins.elemAt matches 0)
              else 0
          ) availableHosts;
          
          maxHost = if hostNumbers != [] 
                   then builtins.foldl' builtins.max 0 hostNumbers
                   else 0;
        in
          if maxHost > 0 then maxHost else fallbackHostCount;

      # Number of hosts to generate
      numHosts = detectAvailableHosts;

      # Generate a single host configuration
      generateHostConfig = hostNum:
        let
          hostname = generateHostname hostNum;
          hostHardwareConfig = hardwareConfigs.${hostname} or null;
        in
          if hostHardwareConfig != null then {
            name = hostname;
            value = nixpkgs.lib.nixosSystem {
              inherit system;
              modules = [
                # Common hardware configuration
                commonHardwareConfig
                
                # Host-specific hardware configuration (fileSystems, swapDevices)
                ({ config, lib, pkgs, ... }: hostHardwareConfig)
                
                # Base system configuration shared by all hosts
                ({ config, pkgs, ... }: {
                  # Set the hostname
                  networking.hostName = hostname;
                  
                  # Basic system configuration
                  system.stateVersion = "23.11"; # Adjust as needed
                  
                  # Add your common configuration here
                  services.openssh.enable = true;
                  networking.firewall.enable = true;
                  
                  # Example: jelly-proxy specific configuration
                  # services.nginx.enable = true;
                  # services.nginx.virtualHosts.${hostname} = {
                  #   # Your proxy configuration
                  # };
                  
                  # You could also add host-specific configuration based on hostname
                  # networking.interfaces.eth0.ipv4.addresses = [{
                  #   address = "192.168.1.${toString (10 + hostNum)}";
                  #   prefixLength = 24;
                  # }];
                })
                
                # Add other common modules here
                # ./modules/common.nix
                # ./modules/jelly-proxy.nix
              ];
              
              # Pass inputs to modules if needed
              specialArgs = { inherit inputs; };
            };
          } else null;

      # Generate all host configurations
      hostConfigs = 
        let
          # Generate list of host numbers from 1 to numHosts
          hostNumbers = builtins.genList (i: i + 1) numHosts;
          
          # Generate configurations for each host
          configs = builtins.map generateHostConfig hostNumbers;
          
          # Filter out null configurations (missing hardware configs)
          validConfigs = builtins.filter (config: config != null) configs;
        in
          builtins.listToAttrs validConfigs;

      # Debug information
      debugInfo = {
        detectedHostCount = numHosts;
        availableHardwareConfigs = builtins.attrNames hardwareConfigs;
        generatedHosts = builtins.attrNames hostConfigs;
        commonHardwareEnabled = true;
      };

    in {
      # NixOS configurations
      nixosConfigurations = hostConfigs;
      
      # Expose debug information
      debug = debugInfo;
      
      # Expose common hardware config for reuse
      lib = {
        inherit commonHardwareConfig;
        inherit hardwareConfigs;
        inherit generateHostname formatHostNumber;
      };
      
      # Helper function to build all hosts
      packages.${system} = {
        all-hosts = nixpkgs.legacyPackages.${system}.writeShellScriptBin "build-all-hosts" ''
          echo "Building ${toString (builtins.length (builtins.attrNames hostConfigs))} hosts..."
          ${builtins.concatStringsSep "\n" (builtins.map (hostname: 
            "echo 'Building ${hostname}...' && nixos-rebuild build --flake .#${hostname}"
          ) (builtins.attrNames hostConfigs))}
        '';

        # Helper to generate hardware config template
        generate-hardware-template = nixpkgs.legacyPackages.${system}.writeShellScriptBin "generate-hardware-template" ''
          if [ -z "$1" ]; then
            echo "Usage: $0 <host-number>"
            echo "Example: $0 6  # generates template for jelly-proxy-06"
            exit 1
          fi
          
          HOST_NUM=$(printf "%02d" "$1")
          echo "Add this to hardwareConfigs in your flake.nix:"
          echo ""
          echo "jelly-proxy-$HOST_NUM = {"
          echo "  fileSystems.\"/\" = {"
          echo "    device = \"/dev/disk/by-uuid/YOUR-ROOT-UUID-HERE\";"
          echo "    fsType = \"ext4\";"
          echo "  };"
          echo "  fileSystems.\"/boot\" = {"
          echo "    device = \"/dev/disk/by-uuid/YOUR-BOOT-UUID-HERE\";"
          echo "    fsType = \"vfat\";"
          echo "  };"
          echo "  swapDevices = ["
          echo "    { device = \"/dev/disk/by-uuid/YOUR-SWAP-UUID-HERE\"; }"
          echo "  ];"
          echo "};"
        '';
      };
      # Development shell
      devShells.${system}.default = nixpkgs.legacyPackages.${system}.mkShell {
        buildInputs = with nixpkgs.legacyPackages.${system}; [
          nixos-rebuild
          nix-output-monitor
        ];
        shellHook = ''
          echo "Jelly Proxy Host Generator"
          echo "========================="
          echo "Detected ${toString numHosts} hosts"
          echo "Available configurations:"
          ${builtins.concatStringsSep "\n" (builtins.map (hostname:
            "echo '  - ${hostname}'"
          ) (builtins.attrNames hostConfigs))}
          echo ""
          echo "Usage:"
          echo "  nixos-rebuild build --flake .#jelly-proxy-01"
          echo "  nix run .#all-hosts  # Build all hosts"
          echo "  nix run .#generate-hardware-template 6  # Generate template for new host"
          echo "  nix eval .#debug --json  # Show debug info"
          echo ""
          echo "Common hardware config includes:"
          echo "  - Boot modules and kernel modules"
          echo "  - CPU microcode updates"
          echo "  - Network DHCP settings"
          echo "  - Platform settings"
        '';
      };
    };
}

    architecture = mkOption {
      type = types.enum [ "intel" "amd" "arm" "auto" ];
      default = "auto";
      description = "CPU architecture to target";
      example = "intel";
    };
    performancePercentage = mkOption {
      type = types.int;
      default = 80;
      description = "Maximum CPU performance percentage (1-100)";
      example = 50;
    };

      type = types.bool;
      default = true;
      description = "Whether to apply the limit at system boot";
    };
  };
  config = mkIf cfg.enable {
    systemd.services.limit-cpu-max-perf = {
      description = "Limit CPU max performance percentage";
      wantedBy = mkIf cfg.applyAtBoot [ "multi-user.target" ];
      serviceConfig = {
        Type = "oneshot";
        ExecStart = let
          # Ensure percentage is within valid range
          percentage = max 1 (min 100 cfg.performancePercentage);
          # Auto-detect architecture if set to auto
          autoDetectScript = ''
            if [ -d "/sys/devices/system/cpu/intel_pstate" ]; then
              ARCH="intel"
            elif [ -d "/sys/devices/system/cpu/amd_pstate" ]; then
              ARCH="amd"
            elif [ -f "/sys/devices/system/cpu/cpufreq/policy0/scaling_max_freq" ]; then
              ARCH="arm"
            else
              echo "Could not auto-detect CPU architecture" >&2
              exit 1
            fi
          '';
          # Create the script based on architecture
          script = if cfg.architecture == "auto" then ''
            #!/bin/sh
            # Auto-detect architecture
            ${autoDetectScript}
            # Apply based on detected architecture
            case "$ARCH" in
              intel)
                ${archImplementations.intel.checkCommand}
                echo ${toString percentage} > ${archImplementations.intel.path}
                ;;
              amd)
                ${archImplementations.amd.checkCommand}
                echo ${toString percentage} > ${archImplementations.amd.path}
                ;;
              arm)
                ${archImplementations.arm.checkCommand}
                ${archImplementations.arm.writeCommand percentage}
                ;;
            esac
          '' else ''
            #!/bin/sh
            # Apply for specific architecture: ${cfg.architecture}
            ${archImplementations.${cfg.architecture}.checkCommand}
            ${if cfg.architecture == "arm" 
              then archImplementations.arm.writeCommand percentage
              else "echo ${toString percentage} > ${archImplementations.${cfg.architecture}.path}"}
          '';
        in
          pkgs.writeShellScript "limit-cpu-perf.sh" script;
        RemainAfterExit = true;
      };

    if [ ! -f "/sys/devices/system/cpu/cpufreq/policy0/scaling_max_freq" ]; then
      echo "ARM CPU frequency scaling not available" >&2

    # For ARM, we need to get the maximum frequency first
