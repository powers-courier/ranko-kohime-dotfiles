mkGlancesService = hostname: {
  "Glances ${hostname}" = {
    href = "http://${hostname}.${vars.tailscale-fqdn}:61208/";
    description = "Glances";
    icon = "si-iterm2";
    widget = {
      type = "glances";
      url = "http://${hostname}.${vars.tailscale-fqdn}:61208/";
      version = 4;
      metric = "info";
      refreshInterval = 5000;
    };
  };
};
