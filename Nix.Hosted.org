environment.systemPackages = with pkgs; [
  (pkgs.callPackage ({ stdenv, fetchFromGitHub, python3 }:
    stdenv.mkDerivation {
      pname = "rffmpeg";
      version = "1.0.2"; # Latest as of now
      src = fetchFromGitHub {
        owner = "joshuaboniface";
        repo = "rffmpeg";
        rev = "v1.0.2";
        sha256 = "sha256-..."; # Run `nix-prefetch-github joshuaboniface rffmpeg v1.0.2`
      };
      buildInputs = [ python3 ];
      installPhase = ''
        mkdir -p $out/bin
        cp rffmpeg.py $out/bin/rffmpeg

** Docker Compose
#+begin_src yaml :noweb-ref jellyfin-docker-compose
version: "3"
services:
  traefik:
    image: traefik:latest
    command:
      - "--providers.docker=true"
      - "--entrypoints.web.address=:8096"
    ports:
      - "8096:8096"
    volumes:
      - "/var/run/docker.sock:/var/run/docker.sock"
    deploy:
      replicas: 3

  jellyfin:
    image: jellyfin/jellyfin:latest
    user: "8096:8096"
    deploy:
      replicas: 1
    labels:
      - "traefik.http.routers.jellyfin.rule=Host(`jellyfin.theatomicass.com`)"
      - "traefik.http.services.jellyfin.loadbalancer.server.port=8096"
    environment:
      - JELLYFIN_DATABASE_URL=postgresql://jellyfin:your-secure-password@192.168.0.24:5432/jellyfin
      - JELLYFIN_FFMPEG_PATH=/usr/local/bin/rffmpeg
    volumes:
      - /Mounts/Videos:/media
      - /Mounts/JellyfinConfig:/config
      - /usr/bin/rffmpeg:/usr/local/bin/rffmpeg:ro
    devices:
      - /dev/dri:/dev/dri
#+end_src
