***** ddrescue disk to file
-b sets sector size (default 512)
-c sets cluster size (default 128)
-d use direct disc access for input file
-J reread latest good sector after every error
-O reopen input file after every read error
-r exit after <n> retry passes (-1=infinity) (default 0)
-R reverse the direction of all passes
-T maximum time since last successful read, in seconds?
-v be verbose (a 2nd -v gives more)
#+begin_src bash
cd_ddrescue()
{
  ddrescue -dOvv -r 5 -b 2048 -T 7200 \
  $CD_ACTIVE_DRIVE $CD_FILENAME.iso $CD_FILENAME.map &&\
  ddrescuelog -d $CD_FILENAME.map &&\
  eject $CD_ACTIVE_DRIVE || cd_rip_error
}
#+end_src

functionCPUSetMaxPerfPct()
{
  # Check if AMD or Intel
  # If AMD, use cpuAMDMaxPerf
  # If Intel, use cpuIntelMaxPerf
  # If neither, exit 1
  if [ -e "$cpuIntelMaxPerf" ]; then
    cpuMaxPerf="$cpuIntelMaxPerf"
  elif [ -e "$cpuAMDMaxPerf" ]; then
    cpuMaxPerf="$cpuAMDMaxPerf"
  else
    echo "Error: $cpuIntelMaxPerf and $cpuAMDMaxPerf do not exist"
    exit 1
  fi

  userRequestedMaxPerf=$(dialogInput "Enter a value between 0 and 100")
  echo $userRequestedMaxPerf > $cpuMaxPerf
}
functionCPUTurboToggle()
{
    if [ -e "$cpuIntelNoTurbo" ]; then
      case "$(cat "$cpuIntelNoTurbo")" in
        0) echo "1" > "$cpuIntelNoTurbo";;
        1) echo "0" > "$cpuIntelNoTurbo";;
        *) echo "Error: $cpuIntelNoTurbo is not 0 or 1"; exit 1;;
      esac
    elif [ -e "$cpuAMDNoTurbo" ]; then
      case "$(cat "$cpuAMDNoTurbo")" in
        0) echo "1" > "$cpuAMDNoTurbo";;
        1) echo "0" > "$cpuAMDNoTurbo";;
        *) echo "Error: $cpuAMDNoTurbo is not 0 or 1"; exit 1;;
      esac
    else
      echo "Error: $cpuIntelNoTurbo and $cpuAMDNoTurbo do not exist"
      exit 1
    fi
}

menu_deptest_grep()
{
  cat $SCRIPTPATH/*/*.func \
    | sed -e 's/\"/./g' -e "s/'[^']*'//g" -e 's/"[^"]*"//g' \
    | sed -e "s/^[[:space:]]*#.*$//" -e "s/\([^\\]\)#[^\"']*$/\1/" \
    | sed -e "s/&&/;/g" -e "s/||/;/g" | tr ";|" "\n\n" \
    | awk '{print $1}' | sort -u | uniq
}

root_perms()
(
if [ "${USERNAME}" != "root" ]
  then
    echo -n "This script must run as root.  Switch to root?  Y/N [y] "
    read YN
    if [ "x${YN}" == "x" ]
        then YN=y
    fi

    if [ "${YN}" == "y" ]
      then
        echo " switching to root"
        sudo ${0} recall
        exit 0
      else
        echo "\n\t Cannot continue, exiting..."
        exit 1
      fi
  else
    echo "\n\t detected root permissions"
fi
)
