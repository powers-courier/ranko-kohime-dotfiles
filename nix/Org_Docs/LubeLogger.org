#+begin_example nix
{ config, lib, pkgs, ... }:

let
  cfg = config.services.LubeLogger;
in
{
  };
  config = lib.mkIf cfg.enable {
    # Delegate to the official nixpkgs module (native .NET service, no Docker)
    services.lubelogger = {
      enable = true;
      inherit (cfg) port openFirewall;

      # Map dataDir (official expects a name under /var/lib/)
      dataDir = if lib.hasPrefix "/" cfg.dataDir
        then lib.removePrefix "/var/lib/" (lib.removeSuffix "/" cfg.dataDir)
        else cfg.dataDir;

      # User's old "environment" maps directly to the official "settings"
      settings = cfg.environment;
    };

    # Extra systemd hardening on top of the official module

    # Ensure the data directory exists with correct permissions
    # (the official module already creates /var/lib/${dataDir}, but this covers custom absolute paths)
    # Optional: re-add your apple-touch overlay if you still need it
    # imports = [ ../lubelogger-apple-touch-overlay.nix ];
  };
}
