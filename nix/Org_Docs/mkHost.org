{ inputs, lib, autoModules, flakeModules, platformModules, nixpkgs, ... }:
{
  <<mkhost-generator-options>>
} @ args:
  let
    isDarwin = platform == "darwin" || lib.hasSuffix "-darwin" system;
    isNixOS  = !isDarwin;
    <<mkhost-generator-platform-selector>>
    <<mkhost-generator-role-modules>>
  in
  if isDarwin then
    inputs.nix-darwin.lib.darwinSystem {
      inherit system;
      specialArgs = { inherit inputs self; };
      modules = lib.flatten [
        (builtins.attrValues autoModules)
        { networking.hostName = hostname; }
        extraModules
      ];
    }
  else
    nixpkgs.lib.nixosSystem {
      inherit system;
      specialArgs = {
        inherit inputs;
      };
      <<mkhost-generator-modules>>
    };
#+end_src
